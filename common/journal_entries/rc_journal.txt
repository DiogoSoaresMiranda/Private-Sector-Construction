hidden_journal = {
    icon = "gfx/interface/icons/event_icons/event_industry.dds"
    
    is_shown_when_inactive = {
        is_ai = yes
        OR = {
            AND = {
                has_global_variable = hidden_journal_main_holder
                OR = {
                    this = global_var:hidden_journal_main_holder
                    global_var:hidden_journal_main_holder = {
                        OR = {
                            is_country_alive = no
                            OR = {
                                is_ai = no
                                NOT = { # Human players eligible only if they conquered all the AIs
                                    any_country = {
                                        is_ai = yes
                                    }
                                }
                            }
                        }
                    }
                }
            }
            NOT = {
                has_global_variable = hidden_journal_main_holder
            }
        }
    }
    
    possible = {
        always = yes
    }
    
    immediate = {
        if = {
            limit = {
                OR = {
                    NOT = {
                        has_global_variable = hidden_journal_main_holder
                    }
                    global_var:hidden_journal_main_holder = {
                        OR = {
                            is_country_alive = no
                            OR = {
                                is_ai = no
                                NOT = {
                                    any_country = {
                                        is_ai = yes
                                    }
                                }
                            }
                        }
                    }
                }
            }
            set_global_variable = {
                name  = hidden_journal_main_holder
                value = this
            }
        }
    }
    
    invalid = {
        OR = {
            is_ai = no
            AND = {
                has_global_variable = hidden_journal_main_holder
                this != global_var:hidden_journal_main_holder
            }
        }
    }
    on_weekly_pulse = {
        effect = {
            trigger_event = {
                on_action = set_construction_start
            }
        }
    }
    can_deactivate = yes
    transferable   = no
    can_revolution_inherit = no
}
