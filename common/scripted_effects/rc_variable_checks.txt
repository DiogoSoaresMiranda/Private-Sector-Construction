check_construction_spending_level = {
    if = {
        limit = {
            NOT = {
                has_variable = construction_spending_level
            }
        }
        if = {
            limit = {
                is_player = yes
            }
            set_variable = {
                name = construction_spending_level
                value = 0.4
            }
        }
        else = {
            set_variable = {
                name = construction_spending_level
                value = 1
            }
        }
    }
}
