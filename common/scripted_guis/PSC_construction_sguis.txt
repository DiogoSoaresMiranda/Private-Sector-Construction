button_1_sgui = {
    scope = country
    effect = {
        set_variable = {
            name = construction_spending_level
            value = 0.2
        }
        custom_tooltip = {
            text = VERY_LOW_CONSTRUCTION_USAGE_PROPORTION_TT
        }
        custom_tooltip = {
            text = VERY_LOW_CONSTRUCTION_BUDGET_COST_TT
        }
    }
    is_valid = {
        has_variable = construction_spending_level
        custom_tooltip = {
            text = CURRENT_DESC
            NOT = {
                var:construction_spending_level = 0.2
            }
        }
    }

}
