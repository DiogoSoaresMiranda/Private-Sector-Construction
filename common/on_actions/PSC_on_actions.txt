set_construction = {
	effect = {
		every_country = {
			trigger_event = {
				on_action = set_construction_country
			}
		}
	}
}

set_construction_country = {
	effect = {
		save_max_construction_spending_values = yes
		set_construction_point_demand = yes
	}
}

set_construction_weekly_on_action = {
    effect = {
        every_country = {
            trigger_event = {
                on_action = set_construction_country
            }
            trigger_event = {
                on_action = set_construction_country
                days = 7
            }
            trigger_event = {
                on_action = set_construction_country
                days = 14
            }
            trigger_event = {
                on_action = set_construction_country
                days = 21
            }
        }
    }
}

on_monthly_pulse = {
    on_actions = {
        set_construction_weekly_on_action
    }
}
